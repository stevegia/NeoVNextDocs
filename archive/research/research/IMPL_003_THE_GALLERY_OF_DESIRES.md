    -- Cleanup tracking
    is_cleaned_up BOOLEAN DEFAULT FALSE
);

-- Indexes
CREATE INDEX idx_staged_inputs_job ON staged_inputs(job_id);
CREATE INDEX idx_staged_inputs_expires ON staged_inputs(expires_at) WHERE NOT is_cleaned_up;
CREATE INDEX idx_staged_inputs_cleanup ON staged_inputs(is_cleaned_up, expires_at);

-- Cleanup function (called by cron job or worker)
CREATE OR REPLACE FUNCTION cleanup_expired_staged_inputs() 
RETURNS INTEGER AS $$
DECLARE
    deleted_count INTEGER;
BEGIN
    -- Delete expired staging records
    DELETE FROM staged_inputs
    WHERE expires_at < NOW() 
      AND NOT is_cleaned_up;
    
    GET DIAGNOSTICS deleted_count = ROW_COUNT;
    
    RETURN deleted_count;
END;
$$ LANGUAGE plpgsql;

```python
"""
Thumbnail extraction handler with fixed silent failure bug.

CHANGES:
1. REMOVED _STUB_PNG gaslighting (no more silent failures)
2. Added ffprobe validation before extraction
3. Configurable success threshold (default: 80%)
4. Explicit exceptions for corrupt/invalid videos
5. Database persistence for thumbnail metadata
"""

import subprocess
import json
from pathlib import Path
from PIL import Image
import shutil
import logging
import os
from cluster_db import ClusterDbClient

logger = logging.getLogger(__name__)

# Configuration
SUCCESS_THRESHOLD = float(os.environ.get("THUMBNAIL_SUCCESS_THRESHOLD", "0.8"))  # 80% default
THUMBNAIL_COUNT = int(os.environ.get("THUMBNAIL_COUNT", "10"))
THUMBNAIL_WIDTH = int(os.environ.get("THUMBNAIL_WIDTH", "320"))
THUMBNAIL_HEIGHT = int(os.environ.get("THUMBNAIL_HEIGHT", "180"))
GCS_THUMBNAIL_BUCKET = os.environ.get("GCS_THUMBNAIL_BUCKET", "gs://neovnext-neovlab-thumbnails")

class VideoValidationError(Exception):
    """Raised when video file is corrupt or invalid."""
    pass

class ThumbnailExtractionError(Exception):
    """Raised when thumbnail extraction fails below threshold."""
    pass

def validate_video_with_ffprobe(file_path: str) -> dict:
    """
    Validate video file using ffprobe.
    
    Args:
        file_path: Path to video file
    
    Returns:
        dict with video metadata (duration, width, height, codec)
    
    Raises:
        VideoValidationError: If file is not a valid video
    """
    cmd = [
        "ffprobe",
        "-v", "error",
        "-select_streams", "v:0",
        "-show_entries", "stream=codec_name,width,height,duration",
        "-show_entries", "format=duration",
        "-of", "json",
        file_path
    ]
    
    try:
        result = subprocess.run(cmd, capture_output=True, text=True, check=True, timeout=30)
        data = json.loads(result.stdout)
        
        # Check for video stream
        if not data.get("streams"):
            raise VideoValidationError(f"No video streams found in {file_path}")
        
        stream = data["streams"][0]
        format_info = data.get("format", {})
        
        # Extract metadata
        codec = stream.get("codec_name", "unknown")
        width = stream.get("width")
        height = stream.get("height")
        duration = stream.get("duration") or format_info.get("duration")
        
        if not width or not height:
            raise VideoValidationError(f"Video dimensions not found in {file_path}")
        
        if not duration:
            raise VideoValidationError(f"Video duration not found in {file_path}")
        
        duration_float = float(duration)
        
        if duration_float <= 0:
            raise VideoValidationError(f"Invalid duration: {duration_float}s")
        
        return {
            "codec": codec,
            "width": int(width),
            "height": int(height),
            "duration_seconds": duration_float
        }
    
    except subprocess.CalledProcessError as e:
        raise VideoValidationError(f"ffprobe failed: {e.stderr}")
    
    except subprocess.TimeoutExpired:
        raise VideoValidationError(f"ffprobe timeout (file may be corrupt)")
    
    except (json.JSONDecodeError, KeyError, ValueError) as e:
        raise VideoValidationError(f"Failed to parse ffprobe output: {e}")

def extract_thumbnail_at_time(
    video_path: str,
    output_path: str,
    timestamp_seconds: float,
    width: int,
    height: int
) -> bool:
    """
    Extract single thumbnail from video at specific timestamp.
    
    Returns:
        True if extraction succeeded, False otherwise
    """
    cmd = [
        "ffmpeg",
        "-y",  # Overwrite output
        "-v", "error",
        "-ss", str(timestamp_seconds),
        "-i", video_path,
        "-vframes", "1",
        "-vf", f"scale={width}:{height}:force_original_aspect_ratio=decrease,pad={width}:{height}:(ow-iw)/2:(oh-ih)/2",
        "-q:v", "2",  # Quality (2 = high)
        output_path
    ]
    
    try:
        subprocess.run(cmd, check=True, capture_output=True, timeout=30)
        
        # Verify file was created and is not empty
        output_file = Path(output_path)
        if not output_file.exists():
            logger.warning(f"ffmpeg did not create output file: {output_path}")
            return False
        
        if output_file.stat().st_size < 500:  # Less than 500 bytes = likely corrupt
            logger.warning(f"Thumbnail file too small ({output_file.stat().st_size} bytes): {output_path}")
            output_file.unlink()  # Delete corrupt file
            return False
        
        # Verify it's a valid image
        try:
            with Image.open(output_path) as img:
                img.verify()
            return True
        except Exception as e:
            logger.warning(f"Invalid image file: {output_path} - {e}")
            output_file.unlink()
            return False
    
    except subprocess.CalledProcessError as e:
        logger.warning(f"ffmpeg extraction failed at {timestamp_seconds}s: {e.stderr.decode()}")
        return False
    
    except subprocess.TimeoutExpired:
        logger.warning(f"ffmpeg timeout at {timestamp_seconds}s")
        return False

def handle(input_path: str, output_dir: str, **kwargs) -> dict:
    """
    Extract thumbnails from video with validation and success threshold.
    
    Args:
        input_path: Path to video file
        output_dir: GCS-mounted output directory
        **kwargs: Contains 'job' dict with media_id, job_id
    
    Returns:
        dict with 'artifacts' (thumbnail files) and 'metadata'
    
    Raises:
        ValueError: If media_id or job_id missing
        VideoValidationError: If video is corrupt or invalid
        ThumbnailExtractionError: If success rate below threshold
    """
    job_info = kwargs.get("job", {})
    media_id = job_info.get("media_id")
    job_id = job_info.get("job_id")
    
    if not media_id or not job_id:
        raise ValueError("Missing media_id or job_id in job context")
    
    logger.info(f"Starting thumbnail extraction for media_id={media_id}, job_id={job_id}")
    
    # STEP 1: Validate video with ffprobe
    logger.info("Validating video with ffprobe...")
    try:
        video_metadata = validate_video_with_ffprobe(input_path)
    except VideoValidationError as e:
        logger.error(f"Video validation failed: {e}")
        raise  # Re-raise exception (NO STUB_PNG GASLIGHTING)
    
    duration = video_metadata["duration_seconds"]
    logger.info(f"Video validated: {duration:.2f}s @ {video_metadata['width']}×{video_metadata['height']}")
    
    # STEP 2: Calculate thumbnail timestamps
    if duration < 1.0:
        # Very short video: extract single thumbnail at midpoint
        timestamps = [duration / 2]
        logger.warning(f"Video too short for {THUMBNAIL_COUNT} thumbnails, extracting 1 at {timestamps[0]:.2f}s")
    else:
        # Evenly spaced thumbnails (skip first/last 5% to avoid black frames)
        start_offset = duration * 0.05
        end_offset = duration * 0.95
        usable_duration = end_offset - start_offset
        
        thumbnail_count = min(THUMBNAIL_COUNT, int(duration))  # Can't extract more than video length
        
        if thumbnail_count == 1:
            timestamps = [duration / 2]
        else:
            interval = usable_duration / (thumbnail_count - 1)
            timestamps = [start_offset + i * interval for i in range(thumbnail_count)]
    
    logger.info(f"Extracting {len(timestamps)} thumbnails at: {[f'{t:.2f}s' for t in timestamps]}")
    
    # STEP 3: Extract thumbnails
    output_path = Path(output_dir)
    output_path.mkdir(parents=True, exist_ok=True)
    
    artifacts = []
    successful_extractions = 0
    
    for idx, timestamp in enumerate(timestamps):
        thumb_filename = f"thumb_{idx:04d}.jpg"
        thumb_path = output_path / thumb_filename
        
        logger.debug(f"Extracting thumbnail {idx+1}/{len(timestamps)} at {timestamp:.2f}s...")
        
        success = extract_thumbnail_at_time(
            video_path=input_path,
            output_path=str(thumb_path),
            timestamp_seconds=timestamp,
            width=THUMBNAIL_WIDTH,
            height=THUMBNAIL_HEIGHT
        )
        
        if success:
            successful_extractions += 1
            
            artifacts.append({
                "type": "thumbnail",
                "path": str(thumb_path),
                "sequence_number": idx,
                "timestamp_seconds": timestamp,
                "size_bytes": thumb_path.stat().st_size
            })
        else:
            logger.warning(f"Failed to extract thumbnail at {timestamp:.2f}s")
    
    # STEP 4: Check success threshold
    success_rate = successful_extractions / len(timestamps)
    
    logger.info(
        f"Thumbnail extraction complete: {successful_extractions}/{len(timestamps)} "
        f"({success_rate*100:.1f}%) succeeded"
    )
    
    if success_rate < SUCCESS_THRESHOLD:
        # FAIL LOUDLY if below threshold
        raise ThumbnailExtractionError(
            f"Thumbnail extraction failed: {success_rate*100:.1f}% success rate "
            f"(threshold: {SUCCESS_THRESHOLD*100:.1f}%)"
        )
    
    # STEP 5: Persist thumbnail metadata to database
    db = ClusterDbClient(os.environ["DATABASE_URL"])
    
    for artifact in artifacts:
        gcs_path = f"{GCS_THUMBNAIL_BUCKET}/{media_id}/{Path(artifact['path']).name}"
        
        db.execute("""
            INSERT INTO thumbnails 
                (media_id, job_id, sequence_number, timestamp_seconds, 
                 thumbnail_path, width_px, height_px, file_size_bytes)
            VALUES (%s, %s, %s, %s, %s, %s, %s, %s)
        """, (
            media_id, job_id, artifact["sequence_number"], artifact["timestamp_seconds"],
            gcs_path, THUMBNAIL_WIDTH, THUMBNAIL_HEIGHT, artifact["size_bytes"]
        ))
    
    logger.info(f"Persisted {len(artifacts)} thumbnail records to database")
    
    return {
        "artifacts": artifacts,
        "metadata": {
            "thumbnail_count": successful_extractions,
            "total_attempted": len(timestamps),
            "success_rate": success_rate,
            "video_duration_seconds": duration,
            "thumbnail_dimensions": f"{THUMBNAIL_WIDTH}×{THUMBNAIL_HEIGHT}"
        }
    }
```

def get_thumbnails_for_media(self, media_id: str) -> list:
    """Retrieve all thumbnails for a media file."""
    with self._cursor() as cursor:
        cursor.execute("""
            SELECT 
                thumbnail_id, media_id, job_id, sequence_number,
                timestamp_seconds, thumbnail_path, width_px, height_px,
                file_size_bytes, generated_at
            FROM thumbnails
            WHERE media_id = %s
            ORDER BY sequence_number ASC
        """, (media_id,))
        
        return [dict(row) for row in cursor.fetchall()]

def insert_staged_input(
    self,
    staging_id: str,
    job_id: str,
    file_path: str,
    expires_hours: int = 2
) -> None:
    """Insert staged input path for job (replaces static dictionary)."""
    with self._cursor() as cursor:
        cursor.execute("""
            INSERT INTO staged_inputs (staging_id, job_id, file_path, expires_at)
            VALUES (%s, %s, %s, NOW() + INTERVAL '%s hours')
        """, (staging_id, job_id, file_path, expires_hours))

def get_staged_input(self, staging_id: str) -> dict:
    """Retrieve staged input path by staging_id."""
    with self._cursor() as cursor:
        cursor.execute("""
            UPDATE staged_inputs
            SET accessed_at = NOW()
            WHERE staging_id = %s AND NOT is_cleaned_up AND expires_at > NOW()
            RETURNING staging_id, job_id, file_path, expires_at
        """, (staging_id,))
        
        row = cursor.fetchone()
        return dict(row) if row else None

def cleanup_expired_staged_inputs(self) -> int:
    """Delete expired staged inputs."""
    with self._cursor() as cursor:
        cursor.execute("""
            DELETE FROM staged_inputs
            WHERE expires_at < NOW() AND NOT is_cleaned_up
        """)
        return cursor.rowcount

using Npgsql;

public class JobOrchestrator
{
    private readonly string _clusterDbConnectionString;

    public JobOrchestrator(IConfiguration config)
    {
        _clusterDbConnectionString = config.GetConnectionString("ClusterDatabase");
    }

    public async Task<string> EnqueueComfyUIJobAsync(
        string inputPath,
        string workflowName,
        Dictionary<string, object> parameters,
        CancellationToken ct = default)
    {
        var stagingId = Guid.NewGuid().ToString();
        var jobId = Guid.NewGuid().ToString();
        
        // Store in database (GOOD)
        await using var conn = new NpgsqlConnection(_clusterDbConnectionString);
        await conn.OpenAsync(ct);
        
        await using var cmd = new NpgsqlCommand(@"
            INSERT INTO staged_inputs (staging_id, job_id, file_path, expires_at)
            VALUES (@staging_id, @job_id, @file_path, NOW() + INTERVAL '2 hours')
        ", conn);
        
        cmd.Parameters.AddWithValue("staging_id", stagingId);
        cmd.Parameters.AddWithValue("job_id", jobId);
        cmd.Parameters.AddWithValue("file_path", inputPath);
        
        await cmd.ExecuteNonQueryAsync(ct);
        
        // Enqueue job to job_queue...
        // (job record includes staging_id in metadata)
        
        return jobId;
    }
    
    public async Task<string?> GetStagedInputPathAsync(string stagingId, CancellationToken ct = default)
    {
        await using var conn = new NpgsqlConnection(_clusterDbConnectionString);
        await conn.OpenAsync(ct);
        
        await using var cmd = new NpgsqlCommand(@"
            UPDATE staged_inputs
            SET accessed_at = NOW()
            WHERE staging_id = @staging_id
              AND NOT is_cleaned_up
              AND expires_at > NOW()
            RETURNING file_path
        ", conn);
        
        cmd.Parameters.AddWithValue("staging_id", stagingId);
        
        var result = await cmd.ExecuteScalarAsync(ct);
        return result?.ToString();
    }
}

import React, { useState, useEffect } from 'react';
import './ThumbnailGallery.css';

interface Thumbnail {
    thumbnailId: string;
    sequenceNumber: number;
    timestampSeconds: number;
    thumbnailPath: string;
    widthPx: number;
    heightPx: number;
}

interface ThumbnailGalleryProps {
    mediaId: string;
}

export const ThumbnailGallery: React.FC<ThumbnailGalleryProps> = ({ mediaId }) => {
    const [thumbnails, setThumbnails] = useState<Thumbnail[]>([]);
    const [isLoading, setIsLoading] = useState(true);
    const [error, setError] = useState<string | null>(null);
    const [selectedThumb, setSelectedThumb] = useState<Thumbnail | null>(null);

    useEffect(() => {
        fetchThumbnails();
    }, [mediaId]);

    const fetchThumbnails = async () => {
        setIsLoading(true);
        setError(null);

        try {
            const response = await fetch(`/api/media/${mediaId}/thumbnails`);

            if (!response.ok) {
                if (response.status === 404) {
                    setThumbnails([]);
                    return;
                }
                throw new Error(`Failed to fetch thumbnails: ${response.statusText}`);
            }

            const data = await response.json();
            setThumbnails(data);
        } catch (err) {
            setError(err instanceof Error ? err.message : 'Failed to load thumbnails');
        } finally {
            setIsLoading(false);
        }
    };

    const formatTimestamp = (seconds: number): string => {
        const mins = Math.floor(seconds / 60);
        const secs = Math.floor(seconds % 60);
        return `${mins}:${secs.toString().padStart(2, '0')}`;
    };

    const getThumbnailUrl = (thumbnail: Thumbnail): string => {
        // Convert GCS path to API endpoint
        return `/api/artifacts/thumbnail/${thumbnail.thumbnailId}`;
    };

    if (isLoading) {
        return <div className="thumbnail-gallery-loading">Loading thumbnails...</div>;
    }

    if (error) {
        return <div className="thumbnail-gallery-error">❌ {error}</div>;
    }

    if (thumbnails.length === 0) {
        return <div className="thumbnail-gallery-empty">No thumbnails available for this video.</div>;
    }

    return (
        <div className="thumbnail-gallery">
            <h3 className="thumbnail-gallery-title">
                Video Thumbnails ({thumbnails.length})
            </h3>

            <div className="thumbnail-grid">
                {thumbnails.map((thumb) => (
                    <div
                        key={thumb.thumbnailId}
                        className={`thumbnail-item ${selectedThumb?.thumbnailId === thumb.thumbnailId ? 'selected' : ''}`}
                        onClick={() => setSelectedThumb(thumb)}
                    >
                        <img
                            src={getThumbnailUrl(thumb)}
                            alt={`Thumbnail at ${formatTimestamp(thumb.timestampSeconds)}`}
                            className="thumbnail-image"
                            loading="lazy"
                        />
                        <div className="thumbnail-overlay">
                            <span className="thumbnail-timestamp">
                                {formatTimestamp(thumb.timestampSeconds)}
                            </span>
                        </div>
                    </div>
                ))}
            </div>

            {selectedThumb && (
                <div className="thumbnail-preview">
                    <div className="preview-header">
                        <h4>Preview</h4>
                        <button onClick={() => setSelectedThumb(null)}>✕</button>
                    </div>
                    <img
                        src={getThumbnailUrl(selectedThumb)}
                        alt="Selected thumbnail"
                        className="preview-image"
                    />
                    <div className="preview-info">
                        <p>Timestamp: {formatTimestamp(selectedThumb.timestampSeconds)}</p>
                        <p>Dimensions: {selectedThumb.widthPx}×{selectedThumb.heightPx}</p>
                    </div>
                </div>
            )}
        </div>
    );
};

/// <summary>
/// Get all thumbnails for a media file.
/// </summary>
[HttpGet("{mediaId}/thumbnails")]
[ProducesResponseType(typeof(List<ThumbnailDto>), StatusCodes.Status200OK)]
[ProducesResponseType(StatusCodes.Status404NotFound)]
public async Task<IActionResult> GetThumbnails(string mediaId, CancellationToken ct = default)
{
    // Query cluster database for thumbnails
    using var conn = new NpgsqlConnection(_clusterDbConnectionString);
    await conn.OpenAsync(ct);

    using var cmd = new NpgsqlCommand(@"
        SELECT thumbnail_id, media_id, sequence_number, timestamp_seconds,
               thumbnail_path, width_px, height_px, file_size_bytes, generated_at
        FROM thumbnails
        WHERE media_id = @media_id
        ORDER BY sequence_number ASC
    ", conn);

    cmd.Parameters.AddWithValue("media_id", mediaId);

    var thumbnails = new List<ThumbnailDto>();

    using var reader = await cmd.ExecuteReaderAsync(ct);
    while (await reader.ReadAsync(ct))
    {
        thumbnails.Add(new ThumbnailDto
        {
            ThumbnailId = reader.GetString(0),
            MediaId = reader.GetString(1),
            SequenceNumber = reader.GetInt32(2),
            TimestampSeconds = reader.GetDouble(3),
            ThumbnailPath = reader.GetString(4),
            WidthPx = reader.GetInt32(5),
            HeightPx = reader.GetInt32(6),
            FileSizeBytes = reader.GetInt32(7),
            GeneratedAt = reader.GetDateTime(8)
        });
    }

    if (thumbnails.Count == 0)
    {
        return NotFound(new { error = $"No thumbnails found for media_id: {mediaId}" });
    }

    return Ok(thumbnails);
}

/// <summary>
/// Serve thumbnail image by thumbnail_id.
/// </summary>
[HttpGet("/api/artifacts/thumbnail/{thumbnailId}")]
[ProducesResponseType(typeof(FileResult), StatusCodes.Status200OK)]
[ProducesResponseType(StatusCodes.Status404NotFound)]
public async Task<IActionResult> GetThumbnailImage(string thumbnailId, CancellationToken ct = default)
{
    // Query for thumbnail path
    using var conn = new NpgsqlConnection(_clusterDbConnectionString);
    await conn.OpenAsync(ct);

    using var cmd = new NpgsqlCommand(@"
        SELECT thumbnail_path FROM thumbnails WHERE thumbnail_id = @thumbnail_id
    ", conn);

    cmd.Parameters.AddWithValue("thumbnail_id", thumbnailId);

    var thumbnailPath = (string?)await cmd.ExecuteScalarAsync(ct);

    if (thumbnailPath == null)
    {
        return NotFound(new { error = $"Thumbnail not found: {thumbnailId}" });
    }

    // Fetch from GCS
    var imageBytes = await _storageService.GetAsync(thumbnailPath, ct);

    return File(imageBytes, "image/jpeg");
}

public class ThumbnailDto
{
    public string ThumbnailId { get; set; } = string.Empty;
    public string MediaId { get; set; } = string.Empty;
    public int SequenceNumber { get; set; }
    public double TimestampSeconds { get; set; }
    public string ThumbnailPath { get; set; } = string.Empty;
    public int WidthPx { get; set; }
    public int HeightPx { get; set; }
    public int FileSizeBytes { get; set; }
    public DateTime GeneratedAt { get; set; }
}
